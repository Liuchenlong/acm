#define M  249997 
#define M1 1000001 
#define M2 0xF0000000 
// BKDR Hash Function  
unsigned int BKDRHash(char*str) 
{ 
    unsigned int seed=131;// 31 131 1313 13131 131313 etc..  
    unsigned int hash=0;
    while(*str)
    {
        hash=hash*seed+(*str++); 
    } 
    return(hash % M); 
}
